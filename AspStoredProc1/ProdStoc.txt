ALTER PROCEDURE [dbo].[SaveID](@ID int,@Name varchar(50))	
AS
BEGIN
	if @Name <> ''
	begin
		if not exists(select * from tblSimple where ID=@ID)
		begin
			insert into tblSimple(Name)
			select  @Name
		end
		else
		begin
			Update tblSimple
			set Name=@Name
			where ID = @ID
		end
	end
	else
	begin
		delete tblSimple where ID = @ID
	end
	
END
